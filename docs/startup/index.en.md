<script>
  const fileList = { release: [], alpha: [] }
  const divList = document.querySelectorAll('div.download-list')
  const githubLink = 'https://github.com/MetaCubeX/mihomo/releases'

  const getLatestRelease = async () => {
    try {
      const link = 'https://api.github.com/repos/MetaCubeX/mihomo/releases/latest'
      const { assets } = await fetch(link).then(r => r.json())
      for (const { name, browser_download_url: url } of assets) {
        fileList.release.push({ name, url })
      }
    } catch (error) {
      console.error('Failed to fetch latest release:', error)
    }
  }

  const getAlphaRelease = async () => {
    try {
      const link = 'https://api.github.com/repos/MetaCubeX/mihomo/releases/tags/Prerelease-Alpha'
      const { assets } = await fetch(link).then(r => r.json())
      for (const { name, browser_download_url: url } of assets) {
        fileList.alpha.push({ name, url })
      }
    } catch (error) {
      console.error('Failed to fetch alpha release:', error)
    }
  }

  const populateDownloadList = () => {
    for (const div of divList) {
      const version = div.getAttribute('data-version')
      const keyword = div.getAttribute('data-keyword')
      const suffix = div.getAttribute('data-suffix')
      const ul = div.querySelector('ul')
      ul.innerHTML = ''

      const files = fileList[version] || []

      for (const { name, url } of files) {
        if (!name.includes(keyword)) continue
        if (!name.includes(suffix)) continue

        const a = document.createElement('a')
        const li = document.createElement('li')
        a.href = url
        a.download = name
        a.innerText = name
        li.appendChild(a)
        ul.appendChild(li)
      }

      if (ul.children.length === 0) {
        ul.innerHTML = `No matching files were found. View all files on GitHub: <a href="${githubLink}" target="_blank">GitHub Release</a>`
      }
    }
  }

  Promise.all([getLatestRelease(), getAlphaRelease()])
    .then(() => {
      populateDownloadList()
    })
    .catch(() => {
      for (const div of divList) {
        const ul = div.querySelector('ul')
        ul.innerHTML = `Loading failed. Download mihomo core binaries from GitHub: <a href="${githubLink}" target="_blank">GitHub Release</a>`
      }
    })
</script>
